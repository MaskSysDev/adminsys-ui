    ```ts title="src/utils/sanitize.ts"
    import { z } from "zod"

    /**
     * Utilitários para sanitização de dados
     *
     * Implementa funções para limpar e padronizar diferentes tipos de dados:
     * - Strings (removendo caracteres especiais)
     * - Emails (padronizando formato)
     * - Nomes (formatando corretamente)
     * - Senhas (preservando caracteres especiais)
     * - Dados de formulário (validação e sanitização)
     * - Respostas de API (validação e sanitização)
     */

    // Lista de caracteres permitidos em regex
    /**
     * Define conjuntos de caracteres permitidos para sanitização de strings.
     * Usado internamente pelas funções de sanitização para construir expressões regulares.
     */
    const ALLOWED_CHARS = {
      // Caracteres básicos
      basic: "a-zA-Z0-9",
      // Espaços e hífens
      spacing: "\\s\\-",
      // Acentos e caracteres especiais da língua portuguesa
      accents: "áàâãéêíóõôúçÁÀÂÃÉÊÍÓÕÔÚÇ^~´`",
      // Caracteres não permitidos (usados em código)
      forbidden: "{}()[]<>=\\+*/\\|&%$#@!?;:\"'",
    }

    /**
     * Sanitiza uma string removendo caracteres especiais e espaços extras
     *
     * Mantém apenas:
     * - Letras e números
     * - Espaços simples
     * - Hífens
     * - Acentos e caracteres especiais da língua portuguesa
     *
     * Exemplos:
     * - "João Paulo" -> "João Paulo"
     * - "José-Maria" -> "José-Maria"
     * - "Café com Leite" -> "Café com Leite"
     * - "Maria da Conceição" -> "Maria da Conceição"
     * - "José da Silva Júnior" -> "José da Silva Júnior"
     */
    export function sanitizeString(value: string): string {
      const allowedChars = `${ALLOWED_CHARS.basic}${ALLOWED_CHARS.spacing}${ALLOWED_CHARS.accents}`

      return value
        .trim()
        .replace(/\s+/g, " ") // Remove espaços extras
        .replace(new RegExp(`[^${allowedChars}]`, "g"), "") // Remove caracteres não permitidos
    }

    /**
     * Sanitiza um email removendo espaços e convertendo para minúsculas
     *
     * Garante que o email esteja em um formato consistente:
     * - Sem espaços no início ou fim
     * - Todo em minúsculas
     */
    export function sanitizeEmail(value: string): string {
      return value.trim().toLowerCase()
    }

    /**
     * Converte um objeto FormData em um objeto JavaScript simples.
     */
    function convertFormDataToObject(formData: FormData): Record<string, string> {
      const obj: Record<string, string> = {}
      formData.forEach((value, key) => {
        obj[key] = value.toString()
      })
      return obj
    }

    /**
     * Sanitiza os campos de string de um objeto, aplicando regras específicas
     * para emails e outros campos de texto, ignorando senhas.
     */
    function sanitizeObjectStrings(data: object): object {
      const sanitizedData = { ...data } as Record<string, unknown>
      for (const [key, value] of Object.entries(sanitizedData)) {
        if (typeof value === "string") {
          if (key.toLowerCase().includes("email")) {
            sanitizedData[key] = sanitizeEmail(value)
          } else if (!key.toLowerCase().includes("password")) {
            sanitizedData[key] = sanitizeString(value)
          }
        }
      }
      return sanitizedData
    }

    /**
     * Sanitiza os dados de entrada de acordo com o schema fornecido.
     * Converte FormData para objeto, sanitiza campos de string (exceto senhas)
     * e valida os dados resultantes contra o schema Zod fornecido.
     *
     * @param data Os dados brutos de entrada (pode ser FormData ou um objeto).
     * @param schema O schema Zod para usar na validação.
     * @returns Os dados sanitizados e validados.
     * @throws Lança um erro se a validação falhar.
     */
    export function sanitizeData<T extends z.ZodType>(
      data: unknown,
      schema: T
    ): z.infer<T> {
      try {
        let processedData = data

        if (processedData instanceof FormData) {
          processedData = convertFormDataToObject(processedData)
        }

        if (typeof processedData === "object" && processedData !== null) {
          processedData = sanitizeObjectStrings(processedData)
        }

        return schema.parse(processedData)
      } catch (error) {
        if (error instanceof z.ZodError) {
          throw new Error(
            `Validation error: ${error.issues.map((e) => e.message).join(", ")}`
          )
        }
        throw error
      }
    }

    /**
     * Sanitiza a resposta da API de acordo com o schema fornecido.
     * Garante que a resposta da API tenha a estrutura e os tipos esperados.
     *
     * @param response A resposta da API a ser validada.
     * @param schema O schema Zod para usar na validação.
     * @returns A resposta validada.
     * @throws Lança um erro se a validação da resposta falhar.
     */
    export function sanitizeResponse<T extends z.ZodType>(
      response: unknown,
      schema: T
    ): z.infer<T> {
      try {
        return schema.parse(response)
      } catch (error) {
        if (error instanceof z.ZodError) {
          throw new Error(
            `Response validation error: ${error.issues.map((e) => e.message).join(", ")}`
          )
        }
        throw error
      }
    }

    /**
     * Sanitiza um nome removendo caracteres especiais e formatando corretamente
     *
     * Aplica as seguintes regras:
     * 1. Remove caracteres especiais (exceto caracteres da língua portuguesa)
     * 2. Remove espaços extras
     * 3. Primeira letra de cada palavra em maiúscula
     *
     * Exemplos:
     * - "joão paulo" -> "João Paulo"
     * - "josé-maria" -> "José-Maria"
     * - "café-com-leite" -> "Café-Com-Leite"
     * - "maria da conceição" -> "Maria Da Conceição"
     * - "josé da silva júnior" -> "José Da Silva Júnior"
     */
    export function sanitizeName(name: string): string {
      // Primeiro sanitiza a string usando a função sanitizeString
      const sanitized = sanitizeString(name)

      return sanitized
        .replace(/^\p{L}/gu, (letter) => letter.toUpperCase()) // Primeira letra maiúscula
        .replace(/\s\p{L}/gu, (letter) => letter.toUpperCase()) // Primeira letra de cada palavra maiúscula
    }

    /**
     * Sanitiza uma senha removendo espaços extras
     *
     * Preserva:
     * - Caracteres especiais
     * - Case-sensitivity
     * - Espaços internos
     *
     * Remove apenas:
     * - Espaços no início e fim
     */
    export function sanitizePassword(password: string): string {
      // Remove espaços extras, mas mantém a senha exatamente como foi digitada
      // para preservar caracteres especiais e case-sensitivity
      return password.trim()
    }
    ```

## Usage

```ts
import { sanitizeData, sanitizeResponse } from "@/utils/sanitize"
import { z } from "zod"

// Exemplo de schema Zod para validação
const userSchema = z.object({
  name: z.string().min(2).max(100),
  email: z.string().email(),
  age: z.number().min(0).optional(),
})

// Sanitizando dados de formulário
const formData = new FormData()
formData.append("name", " João   da Silva ")
formData.append("email", "User@emIL.com    ")
formData.append("age", "30")

const sanitizedUser = sanitizeData(formData, userSchema)
console.log(sanitizedUser)

```
