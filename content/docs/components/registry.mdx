## Create Registry

<Steps>
  <Step>
    ### Create Component

    ```tsx title="src/registry/components/shared/alert/alert-info.tsx"
    import { InfoIcon } from "lucide-react"

    import { Alert, AlertDescription, AlertTitle } from "@/components/ui/alert"

    export type AlertInfoProps = {
      title: string
      description?: string
    }

    export function AlertInfo({ title, description }: AlertInfoProps) {
      return (
        <Alert
          className="border-0 bg-blue-200 text-blue-600 dark:bg-blue-950"
          variant="default"
        >
          <AlertTitle className="flex items-center gap-2">
            <InfoIcon className="size-6" />
            {title}
          </AlertTitle>
          {description && (
            <AlertDescription className="text-muted-foreground">
              {description}
            </AlertDescription>
          )}
        </Alert>
      )
    }

    ```
  </Step>
  <Step>
    ### Registry Component

    ```tsx title="src/registry/registry-components.ts"
    import type { Registry } from "shadcn/schema"

    /**
     * Este arquivo define os metadados para os componentes de UI do projeto.
     * Cada objeto no array representa um componente e suas propriedades, como nome, tipo, descrição,
     * autor, dependências e caminhos de arquivo. Essas informações são usadas pelo processo
     * de construção do registro para gerar arquivos como `index.tsx` e `registry.autogenerated.json`.
     */
    export const components: Registry["items"] = [
      {// [!code ++]
        name: "alert-info",// [!code ++]
        type: "registry:component",// [!code ++]
        description: "A alert-info component.",// [!code ++]
        title: "alert-info",// [!code ++]
        author: "<author> <<authorUrl>>",// [!code ++]
        registryDependencies: ["alert"],// [!code ++]
        files: [// [!code ++]
          {// [!code ++]
            path: "components/shared/alert/alert-info.tsx",// [!code ++]
            type: "registry:component",// [!code ++]
            target: "src/components/shared/alert/alert-info.tsx",// [!code ++]
          },// [!code ++]
        ],// [!code ++]
        docs: "<baseUrl>/docs/components/shared/alert#alert-info",// [!code ++]
      },// [!code ++]
    ]

    ```
  </Step>
</Steps>

    ```tsx title="src/registry/registry-examples.ts"
    import type { Registry } from "shadcn/schema"

    /**
     * Este arquivo define os metadados para os componentes de exemplo do projeto.
     * Cada objeto no array representa um exemplo e suas propriedades, como nome, tipo,
     * dependências de registro e caminhos de arquivo. Essas informações são usadas pelo processo
     * de construção do registro para gerar arquivos como `index.tsx` e `registry.autogenerated.json`,
     * permitindo a demonstração e visualização dos componentes principais.
     */
    export const examples: Registry["items"] = [
      {// [!code ++]
        name: "alert-info-demo",// [!code ++]
        type: "registry:example",// [!code ++]
        registryDependencies: ["<registryBaseUrl>/alert-info.json"],// [!code ++]
        files: [// [!code ++]
          {// [!code ++]
            path: "examples/components/shared/alert/alert-info-demo.tsx",// [!code ++]
            type: "registry:example",// [!code ++]
          },// [!code ++]
        ],// [!code ++]
    ```
